```

K.<theta_K> = NumberField(x^6 - 18*x^4 - 24*x^3 + 27*x^2 + 36*x - 6)
fp = K.fractional_ideal((2, -7/44*theta_K^5 + 19/44*theta_K^4 +

print(fp.norm())               # yields 2

v = K.valuation(fp)            # raises AssertionError
```
