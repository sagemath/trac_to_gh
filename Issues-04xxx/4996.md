```
## We want the debug and optimization flags, since debug symbols are so useful, etc.

env.Append( CXXFLAGS="-O2 -g -m64" )
env.Append( LINKFLAGS="-m64 -single_module -flat_namespace -undefined dynamic_lookup" )
```
