```diff
-    url = "http://dynamic.xkcd.com/api-0/jsonp/comic/{}".format(n)

+        # default to last comic
+        url = "http://xkcd.com/info.0.json"
+    else:
+        url = "https://xkcd.com/{}/info.0.json".format(n)
```
