```diff
-        if p is not None: return p
+        if p is not None:
+            return p
```

or

```diff
-        try: return L.base_extend(ring)(data)
-        except TypeError: pass
-        except ValueError as ex:
-            if str(ex).startswith("Cannot coerce"): pass
+        try:
+            return L.base_extend(ring)(data)

+            pass
```
