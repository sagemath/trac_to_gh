```
u,v = var('u,v')
def cf(u,v): return sin(u+v/2)**2
P = parametric_plot3d((cos(u), sin(u)+cos(v), sin(v)),
(u,0,2*pi), (v,-pi,pi), color=(cf,colormaps.PiYG), plot_points=[60,60])
(P+sphere()).show(viewer='threejs')
```
