def get_polytope():
    q = MixedIntegerLinearProgram(maximization=False, solver='Coin')
    w = q.new_variable(real=True, nonnegative=True) 
    q.add_constraint(w[0]==0)

def fill_memory(n):
    for i in xrange(n):
        P = get_polytope()
