def random_int_upto(n):
    """
    Returns a random integer x satisfying 0 <= x < n.

    EXAMPLES:
        sage: x = random_int_upto(10) 
        sage: x >= 0
        True
        sage: x < 10
        True
    """
    return floor(n * random())
