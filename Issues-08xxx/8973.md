```

+++ b/surf-1.1/surf/draw/SurfaceCalc.cc 2010-05-15 17:48:38.663596876 +0200
@@ -38,6 +38,7 @@
 #include <IO.h>
 
 #include <math.h>
+#include <cstring>
 
 extern double Y_AXIS_LR_ROTATE;

--- a/surf/image-formats/ByExtension.cc 2003-04-14 17:12:37.000000000 +0200
```
