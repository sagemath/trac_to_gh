```
sage: a=Matrix(FiniteField(5),
....: [[2,0,0],
....: [0,3,0],
....: [0,0,1]])
sage: 
sage: b=Matrix(FiniteField(5),
....: [[0,1,0],
....: [4,0,0],
....: [0,0,1]])
sage: 
sage: c=Matrix(FiniteField(5),
....: [[1,0,0],
....: [0,1,0],
....: [0,1,1]])
sage: 
sage: d=Matrix(FiniteField(5),
....: [[1,0,0],
....: [0,1,0],
....: [1,0,1]])
sage: 
sage: G = MatrixGroup([a,b,c,d])
sage: G.center()
```
