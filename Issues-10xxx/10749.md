The fix is the following change:

```

--- a/spkg-install
+++ b/spkg-install
@@ -43,7 +43,7 @@
     exit 1
 fi
 
-if [ $UNAME = "CYGWIN" ]
+if [ $UNAME = "CYGWIN" ]; then
   rm -rf "$SAGE_LOCAL"/lib/libpng12.dll.a
   rm -rf "$SAGE_LOCAL"/lib/libpng12.la

```
