```diff
diff -r 3d7bb3a7774e sage-build

+++ b/sage-build

     exit 1
 fi
 
+# Check whether this seems to a 64-bit build:
+source "$SAGE_LOCAL"/bin/sage-check-64
+
+if [ $? -ne 0 ]; then
+    exit 1

+
```
