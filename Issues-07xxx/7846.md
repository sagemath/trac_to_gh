Which is better

```python
import os, shutil

def ignore_nonexistent_files(curdir, dirlist):
    ignore = []
    for x in dirlist:
        if not os.path.exists(os.path.join(curdir, x)):
            ignore.append(x)
    return ignore

# This:
shutil.copytree('foo', 'bar', ignore=ignore_nonexistent_files)
# Or this:
shutil.copytree('foo', 'bar', symlinks=True)
```
