```

[curl-7.84.0]

[curl-7.84.0]   --with-bearssl

[curl-7.84.0]   --with-mbedtls
[curl-7.84.0]   --with-nss

[curl-7.84.0]   --with-secure-transport
[curl-7.84.0]   --with-wolfssl
```
