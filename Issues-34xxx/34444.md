             sage: I._groebner_cover()

-            [(2*a + 3)*z + (a + 9), (12*a + 18)*y + (-a^2 - 23*a - 36), (4*a + 6)*x + (-a^2 - a + 12)],
-            [[[0], [[(2*a + 3)]]]]],
-            [[1], [1], [[[(2*a + 3)], [[1]]]]]]
-
+             [(2*a + 3)*z + (a + 9), (12*a + 18)*y + (-a^2 - 23*a - 36), (4*a + 6)*x + (-a^2 - a + 12)],
+             [[[0], [[(2*a + 3)]]]]],
+             [[1], [1], [[[(2*a + 3)], [[1]]]]]]
         """
         from sage.rings.fraction_field import FractionField_generic
         F = self.base_ring()
