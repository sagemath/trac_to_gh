sage: Asy.<n> = AsymptoticRing('n^QQbar', ZZ)
sage: BGG = MonomialGrowthGroup.factory(RBF, 'n', extend_by_non_growth_group=True)
sage: BAsy = AsymptoticRing(BGG, ZZ)
sage: BAsy
