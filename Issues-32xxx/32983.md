always work nicely with the coercion framework.

sage: K.<a> = QQ.extension(f)
sage: L.<b> = K.extension(x^2 - 11)
sage: q = L.prime_above(11)
sage: Fq = q.residue_field()
sage: R.<x1> = Fq[]
sage: Fq.0 * x1
