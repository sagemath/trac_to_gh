```
vars = ",".join([ 'c%s'%i for i in range(1, 23)])
BR  = QQ[ 't,'+vars ].fraction_field()
t = BR.gens()[0]
MS = MatrixSpace(BR, 3, 3)

def x(alpha, k, c):
    m = MS(1)
    if alpha == 'a1':
        m[0,1] = BR(c*t^k)
    elif alpha == 'a2':
        m[1,2] = BR(c*t^k)
    elif alpha == 'phi':
        m[0,2] = BR(c*t^k)
    elif alpha == '-a1':
        m[1,0] = BR(c*t^k)
    elif alpha == '-a2':
        m[2,1] = BR(c*t^k)
    elif alpha == '-phi':
        m[2,0] = BR(c*t^k)
    return m
    
    
def n(alpha, k, c):
    if alpha[0] == '-':
        minusalpha = alpha[1:]
    else:
        minusalpha = "-"+alpha
    return x(alpha, k, c)*x(minusalpha,-k,-1/c)*x(alpha, k, c)
    
n0 = n('-phi', 1, 1)
```
