```
-        g = self.__rnf.rnfeltabstorel(pari(f))
+        if self.__K.degree() == 1:
+            g = -1*self.__rnf[0][0]*f[1] + f[0]
+        else:
+            g = self.__rnf.rnfeltabstorel(pari(f))
