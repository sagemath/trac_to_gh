but that's what spkg-check already does, no?

```
#!/usr/bin/env bash

if [ "$SAGE_LOCAL" = "" ]; then
   echo >&2 "SAGE_LOCAL undefined ... exiting";
   echo >&2 "Maybe run 'sage -sh'?"
   exit 1
fi

cd nauty*

# runalltests doesn't exit with a zero status.
# So we check $fails instead.

$MAKE checks
source ./runalltests

if [ "${fails}" -ne "0" ]; then
   echo >&2 "Error checking nauty."
   exit 1
fi
```
