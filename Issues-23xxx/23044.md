def strip_u_prefix(s, q):
    prefix = 'u' + q
    initial = s.split(prefix)
    parity = 0
    final = []
    for part in initial:
        parity += part.count(q)
        final.append(part)
        if parity % 2:
            final.append(prefix)
        else:
            final.append(q)
        parity += 1
    return u''.join(final[:-1])
