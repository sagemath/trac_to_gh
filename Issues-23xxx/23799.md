I meant

```
packages = ['fflas-ffpack', 'givaro', 'linbox', 'gsl', 'singular']
ans = {}
for pkg in packages:
    pc = pkgconfig.parse(pkg)
    name = pkg.replace('-', '').upper()

    ans['%s_LIBDIR' %name] = pc['library_dirs']

return ans
```
