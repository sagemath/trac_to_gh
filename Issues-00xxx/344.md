```
# setup
f = conway_polynomial(2,63)
K.<a> = GF(2**63, name='a', modulus=f)
n = f.degree()
m = 3;
e = (2^n - 1) / (2^m - 1)
c = a^e
conway = conway_polynomial(2,m)
```

```
# element 1
print conway(c)   # says 0
print type(c)
print parent(c)
print c in K          # says True
```

```
# element 2
print K(0)      # says 0
print type(K(0))
print parent(K(0))
```

```
print conway(c) == K(0)    # says False???
```
