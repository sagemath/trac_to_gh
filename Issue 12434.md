-        "$SAGE_ROOT"/spkg/pipestatus \
-            "sage-spkg $OPTINFO $OPTF $OPTS $OPTC '$PKG' 2>&1" \
-            "(trap '' SIGINT; tee -a '$SAGE_ROOT/install.log' '$SAGE_LOGS/$PKG_NAME.log')"
+       if [ -n $OPTINFO ]; then
+            sage-spkg "$OPTINFO" "$PKG" 2>&1
