```python
def groebner_basis_libgiac(gens, epsilon=0, prot=False, <other args>):
   from giacpy import libgiac
   F = libgiac(gens)
   giacsettings.proba_epsilon=epsilon

   B = F.gbasis([P.gens()],'revlex')
```
