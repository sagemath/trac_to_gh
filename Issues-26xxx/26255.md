```python
        V = self.index_set()
        S = self.simple_reflections()
        E = []
        for i,j in combinations(V, 2):
            o = (S[i]*S[j]).order()
            if o >= 3:
                E.append((i,j,o))
        return Graph([V,E], format='vertices_and_edges', immutable=True)
```
