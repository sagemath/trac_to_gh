from sage.misc.sageinspect import sage_getargspec
props = ['is_distributive', 'is_self_dual', 'is_modular']
L = posets.RandomLattice(10, 0.99)
for p in props:
    f = attrcall(p)
    if 'certificate' in sage_getargspec(getattr(L, p)).args:
