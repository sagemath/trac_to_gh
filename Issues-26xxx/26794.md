def check_bad5(n):
    """                                                                                                                                               
    sage: check_bad5(100000)                                                                                                                          
    """

    for i in range(n):
        x = 0 in G.breadth_first_search(0)
        if i % 10000 == 0:
            print get_memory_usage()
