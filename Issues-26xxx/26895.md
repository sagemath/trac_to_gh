....: assume(b, 'real')
....: assume(c, 'real')

2*a == 0
sage: (2*a==2*b).simplify_full() # should be a==b
2*a == 2*b
sage: assume(c != 0)
....: (c*a==c*b).simplify_full() # should be a==b
a*c == b*c
sage: (a/c==b/c).simplify_full() # should be a==b
a/c == b/c
