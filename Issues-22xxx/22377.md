set_random_seed(42)

    L = Posets.RandomLattice(randint(0, 30), 0.99)

        print("Bug 1")
        break
else:
    print("No bug 1")
for i in range(100):
    L = Posets.RandomLattice(randint(0, 30), 0.99)
    a, b = L.is_cosectionally_complemented(certificate=True)
